1 / 1

Un analyseur syntaxique, également appelé parseur, est un outil qui permet d'analyser la syntaxe d'un code source et de construire un arbre syntaxique. L'analyse syntaxique est une étape cruciale dans la compilation d'un programme, car elle permet de vérifier la validité du code et de le transformer en une structure de données abstraite qui peut être traitée par la suite.

L'analyseur syntaxique travaille en général en étroite collaboration avec l'analyseur lexical, qui se charge de découper le code source en tokens ou lexèmes. Les tokens sont ensuite transmis à l'analyseur syntaxique, qui va construire l'arbre syntaxique en appliquant les règles de la grammaire du langage.

Il existe plusieurs approches pour l'analyse syntaxique, mais la plus courante est l'analyse descendante, qui consiste à partir du symbole de départ de la grammaire et à appliquer récursivement les règles de la grammaire jusqu'à reconnaître l'expression à analyser. Cette approche est souvent mise en œuvre avec une pile (stack) pour gérer la récursivité.

L'analyseur syntaxique peut également être accompagné d'un analyseur sémantique, qui se charge de vérifier la cohérence et la validité sémantique du code.

    TIMES

    LPAREN
    RPAREN
    INT
    EOF

type Token struct {
    Type  TokenType
    Value string
}

type Lexer struct {
    input        string
    position     int
    currentToken Token
}

func NewLexer(input string) *Lexer {
    l := &Lexer{
        input:    input,
        position: 0,
    }
    l.getNextToken()
    return l
}

func (l *Lexer) getNextToken() {
    if l.position >= len(l.input) {
        l.currentToken = Token{Type: EOF, Value: ""}
        return
    }

    currentChar := l.input[l.position]

    if currentChar == '+' {
        l.currentToken = Token{Type: PLUS, Value: string(currentChar)}
        l.position++
        return
    }

    if currentChar == '-' {
        l.currentToken = Token{Type: MINUS, Value: string(currentChar)}
        l.position++
        return
    }

    if currentChar == '*' {
        l.currentToken = Token{Type: TIMES, Value: string(currentChar)}
        l.position++
        return
    }

    if currentChar == '/' {
        l.currentToken = Token{Type: DIV, Value: string(currentChar)}
        l.position++
        return
    }

    if currentChar == '(' {
        l.currentToken = Token{Type: LPAREN, Value: string(currentChar)}
        l.position++
        return
    }

    if currentChar == ')' {
        l.currentToken = Token{Type: RPAREN, Value: string(currentChar)}
        l.position++
        return
    }

    if unicode.IsDigit(rune(currentChar)) {
        value := ""
        for l.position < len(l.input) && unicode.IsDigit(rune(l.input[l.position])) {
            value += string(l.input[l.position])
            l.position++
        }
        l.currentToken = Token{Type: INT, Value: value}
        return
    }

    panic(fmt.Sprintf("Invalid character: %v", string(currentChar)))
}

import java.util.regex.*;

public class Lexer {
    private final String input;
    private int currentPos;
    private Token currentToken;

    public Lexer(String input) {
        this.input = input;
        this.currentPos = 0;
        this.currentToken = null;
    }

    public Token getNextToken() throws LexicalException {
        if (currentPos >= input.length()) {
            return new Token(TokenType.EOF, "");
        }
        
        String remainingInput = input.substring(currentPos);
        
        // Match integer literal
        Pattern integerPattern = Pattern.compile("^\\d+");
        Matcher integerMatcher = integerPattern.matcher(remainingInput);
        if (integerMatcher.find()) {
            String matchedInteger = integerMatcher.group();
            currentPos += matchedInteger.length();
            return new Token(TokenType.INTEGER, matchedInteger);
        }

        // Match operators
        if (remainingInput.startsWith("+")) {
            currentPos++;
            return new Token(TokenType.PLUS, "+");
        } else if (remainingInput.startsWith("-")) {
            currentPos++;
            return new Token(TokenType.MINUS, "-");
        } else if (remainingInput.startsWith("*")) {
            currentPos++;
            return new Token(TokenType.MULTIPLY, "*");
        } else if (remainingInput.startsWith("/")) {
            currentPos++;
            return new Token(TokenType.DIVIDE, "/");
        } else if (remainingInput.startsWith("(")) {
            currentPos++;
            return new Token(TokenType.OPEN_PAREN, "(");
        } else if (remainingInput.startsWith(")")) {
            currentPos++;
            return new Token(TokenType.CLOSE_PAREN, ")");
        }

        throw new LexicalException("Unexpected character: " + remainingInput.charAt(0));
    }
}

Cette implémentation utilise des expressions régulières pour détecter les littéraux d'entiers et des correspondances de caractères pour détecter les opérateurs et les parenthèses. Si un caractère inattendu est rencontré, une exception LexicalException est levée. Notez que cette implémentation ne prend pas en compte les espaces ou les caractères de fin de ligne.

import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

public class AnalyseurLexical {
    private String input;
    private List<String> tokens;
    private List<String> separators = Arrays.asList("+", "-", "*", "/", "(", ")", "%");

    public AnalyseurLexical(String input) {
        this.input = input;
        this.tokens = new ArrayList<>();
        tokenize();
    }

    private void tokenize() {
        String token = "";
        for (int i = 0; i < input.length(); i++) {
            char c = input.charAt(i);
            if (Character.isDigit(c)) {
                token += c;
            } else if (separators.contains(String.valueOf(c))) {
                if (!token.isEmpty()) {
                    tokens.add(token);
                    token = "";
                }
                tokens.add(String.valueOf(c));
            } else if (Character.isWhitespace(c)) {
                if (!token.isEmpty()) {
                    tokens.add(token);
                    token = "";
                }
            } else {
                System.out.println("Erreur : caractère inconnu " + c);
                System.exit(1);
            }
        }
        if (!token.isEmpty()) {
            tokens.add(token);
        }
    }

    public List<String> getTokens() {
        return tokens;
    }

    public static void main(String[] args) {
        AnalyseurLexical lexical = new AnalyseurLexical("(4 + 7 / 3) - 5");
        System.out.println(lexical.getTokens());
    }
}

        if (T() && Ep()) {
            return true;
        } else {
            return false;
        }
    }

    private static boolean Ep() {
        if (index < input.length() && (input.charAt(index) == '+' || input.charAt(index) == '-')) {
            index++;
            if (T() && Ep()) {
                return true;
            } else {
                return false;
            }
        } else {
            return true;
        }
    }

    private static boolean T() {
        if (F() && Tp()) {
            return true;
        } else {
            return false;
        }
    }

    private static boolean Tp() {
        if (index < input.length() && (input.charAt(index) == '*' || input.charAt(index) == '/' || input.charAt(index) == '%')) {
            index++;
            if (F() && Tp()) {
                return true;
            } else {
                return false;
            }
        } else {
            return true;
        }
    }

    private static boolean F() {
        if (index < input.length() && input.charAt(index) == '(') {
            index++;
            if (E() && index < input.length() && input.charAt(index) == ')') {
                index++;
                return true;
            } else {
                return false;
            }
        } else if (N()) {
            return true;
        } else {
            return false;
        }
    }

    private static boolean N() {
        if (index < input.length() && Character.isDigit(input.charAt(index))) {
            index++;
            while (index < input.length() && Character.isDigit(input.charAt(index))) {
                index++;
            }
            return true;
        } else {
            return false;
        }
